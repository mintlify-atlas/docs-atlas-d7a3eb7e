## Overview

The Clubs API allows you to manage club information in the MTB Backend system. Each club can have a name, logo, cover image, history, mission, and contact details.

<Note>
This endpoint supports draft and publish functionality. Use the `publishedAt` field to control publication status.
</Note>

---

## List All Clubs

<api-endpoint method="GET" path="/api/clubs" />

Retrieve a list of all clubs with optional filtering, sorting, and pagination.

### Query Parameters

<ParamField query="filters" type="object" optional>
  Filter clubs by field values
</ParamField>

<ParamField query="sort" type="string" optional>
  Sort clubs by field(s). Example: `nombreClub:asc`
</ParamField>

<ParamField query="pagination[page]" type="number" optional default="1">
  Page number for pagination
</ParamField>

<ParamField query="pagination[pageSize]" type="number" optional default="25">
  Number of items per page
</ParamField>

<ParamField query="populate" type="string" optional>
  Populate relations. Use `*` for all relations or specify fields like `logo,imagenPortada`
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of club objects
  
  <Expandable title="Club Object">
    <ResponseField name="id" type="number">
      Unique identifier for the club
    </ResponseField>
    
    <ResponseField name="attributes" type="object">
      <Expandable title="Attributes">
        <ResponseField name="nombreClub" type="string">
          Name of the club
        </ResponseField>
        
        <ResponseField name="logo" type="object">
          Club logo media object
        </ResponseField>
        
        <ResponseField name="imagenPortada" type="object">
          Cover image media object
        </ResponseField>
        
        <ResponseField name="historia" type="array">
          Club history in blocks format
        </ResponseField>
        
        <ResponseField name="mision" type="array">
          Club mission statement in blocks format
        </ResponseField>
        
        <ResponseField name="emailConacto" type="string">
          Contact email address
        </ResponseField>
        
        <ResponseField name="telefonoContacto" type="string">
          Contact phone number
        </ResponseField>
        
        <ResponseField name="direccion" type="string">
          Physical address of the club
        </ResponseField>
        
        <ResponseField name="publishedAt" type="datetime">
          Publication timestamp (null if draft)
        </ResponseField>
        
        <ResponseField name="createdAt" type="datetime">
          Creation timestamp
        </ResponseField>
        
        <ResponseField name="updatedAt" type="datetime">
          Last update timestamp
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata
  
  <Expandable title="Meta Object">
    <ResponseField name="pagination" type="object">
      <ResponseField name="page" type="number">
        Current page number
      </ResponseField>
      <ResponseField name="pageSize" type="number">
        Number of items per page
      </ResponseField>
      <ResponseField name="pageCount" type="number">
        Total number of pages
      </ResponseField>
      <ResponseField name="total" type="number">
        Total number of clubs
      </ResponseField>
    </ResponseField>
  </Expandable>
</ResponseField>

### Example Request

<CodeGroup>
```bash cURL
curl -X GET "https://api.example.com/api/clubs?populate=*&sort=nombreClub:asc" \
  -H "Content-Type: application/json"
```

```javascript JavaScript
const response = await fetch('https://api.example.com/api/clubs?populate=*&sort=nombreClub:asc', {
  method: 'GET',
  headers: {
    'Content-Type': 'application/json'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.example.com/api/clubs',
    params={'populate': '*', 'sort': 'nombreClub:asc'},
    headers={'Content-Type': 'application/json'}
)
data = response.json()
```
</CodeGroup>

## Create a Club

<api-endpoint method="POST" path="/api/clubs" />

Create a new club entry.

### Request Body

<ParamField body="data" type="object" required>
  <Expandable title="Club Data">
    <ParamField body="nombreClub" type="string" optional>
      Name of the club
    </ParamField>
    
    <ParamField body="logo" type="number" optional>
      ID of the uploaded logo media file
    </ParamField>
    
    <ParamField body="imagenPortada" type="number" optional>
      ID of the uploaded cover image media file
    </ParamField>
    
    <ParamField body="historia" type="array" optional>
      Club history in blocks format
    </ParamField>
    
    <ParamField body="mision" type="array" optional>
      Club mission statement in blocks format
    </ParamField>
    
    <ParamField body="emailConacto" type="string" optional>
      Contact email address
    </ParamField>
    
    <ParamField body="telefonoContacto" type="string" optional>
      Contact phone number
    </ParamField>
    
    <ParamField body="direccion" type="string" optional>
      Physical address of the club
    </ParamField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="data" type="object">
  The created club object
</ResponseField>

### Example Request

<CodeGroup>
```bash cURL
curl -X POST "https://api.example.com/api/clubs" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{
    "data": {
      "nombreClub": "Nuevo Club MTB",
      "emailConacto": "nuevo@clubmtb.com",
      "telefonoContacto": "+56987654321",
      "direccion": "Calle Nueva 456, Valparaíso",
      "historia": [
        {
          "type": "paragraph",
          "children": [{"type": "text", "text": "Historia del nuevo club..."}]
        }
      ],
      "mision": [
        {
          "type": "paragraph",
          "children": [{"type": "text", "text": "Nuestra misión es..."}]
        }
      ]
    }
  }'
```

```javascript JavaScript

  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_TOKEN'
```
